
App.js
```
jsx
import React, { useState } from 'react';
import { View, Text, TextInput, Button } from 'react-native';
import TeamMaker from './TeamMaker';

const App = () => {
  const [names, setNames] = useState('');
  const [teamSize, setTeamSize] = useState(0);
  const [teams, setTeams] = useState([]);

  const handleShuffle = () => {
    const teams = TeamMaker.shuffleAndAssignTeams(names.split(','), teamSize);
    setTeams(teams);
  };

  return (
    <View>
      <Text>Team Maker</Text>
      <TextInput
        placeholder="Enter names/numbers separated by commas"
        value={names}
        onChangeText={(text) => setNames(text)}
      />
      <TextInput
        placeholder="Enter team size"
        value={teamSize.toString()}
        onChangeText={(text) => setTeamSize(parseInt(text))}
        keyboardType="numeric"
      />
      <Button title="Shuffle & Assign Teams" onPress={handleShuffle} />
      {teams.map((team, index) => (
        <Text key={index}>Team {index + 1}: {team.join(', ')}</Text>
      ))}
    </View>
  );
};

export default App;
```

TeamMaker.js
```
const shuffleAndAssignTeams = (names, teamSize) => {
  const shuffledNames = names.sort(() => Math.random() - 0.5);
  const teams = [];
  for (let i = 0; i < shuffledNames.length; i += teamSize) {
    teams.push(shuffledNames.slice(i, i + teamSize));
  }
  return teams;
};

export default { shuffleAndAssignTeams };
```
